## What a TurboModule Is

A TurboModule is a **native module for the New Architecture**. It replaces the old `ReactContextBaseJavaModule` pattern with a Codegen-driven, type-safe, JSI-backed implementation.

The mental model for an Android developer:

| Old Native Module | TurboModule |
|-------------------|-------------|
| `ReactContextBaseJavaModule` | Kotlin class implementing a Codegen-generated interface |
| `@ReactMethod` annotation | Method defined in the TypeScript spec |
| Async only (Promise/Callback) | Synchronous or async |
| JSON serialised arguments | Direct JS value types via JSI |
| Registered at startup (all modules) | Loaded on demand (lazy) |
| No type safety end-to-end | TypeScript spec ŌåÆ Codegen ŌåÆ Kotlin interface ŌĆö fully typed |

## Step 1: The TypeScript Spec

The spec is the **contract** between JavaScript and Kotlin. It is a TypeScript file that Codegen reads to generate all the C++/JNI glue.

Create `src/specs/NativeDeviceInfoModule.ts`:

```typescript
// src/specs/NativeDeviceInfoModule.ts
import type { TurboModule } from 'react-native';
import { TurboModuleRegistry } from 'react-native';

export interface Spec extends TurboModule {
  // Synchronous ŌĆö returns immediately
  getDeviceModel(): string;
  getAndroidVersion(): number;

  // Async ŌĆö returns a Promise
  getBatteryLevel(): Promise<number>;
  getNetworkType(): Promise<'wifi' | 'cellular' | 'none'>;

  // With parameters
  readFile(path: string): Promise<string>;
  writeFile(path: string, content: string): Promise<boolean>;

  // Callback-based (use Promise when possible)
  subscribeToOrientationChanges(
    callback: (orientation: 'portrait' | 'landscape') => void
  ): void;
}

export default TurboModuleRegistry.getEnforcing<Spec>('NativeDeviceInfoModule');
```

### Type rules for specs

The Codegen supports a limited set of types. This is intentional ŌĆö every type maps cleanly to both TypeScript and the native layer:

| TypeScript spec type | Kotlin type | Notes |
|---------------------|-------------|-------|
| `string` | `String` | |
| `number` | `Double` | All JS numbers are Double |
| `boolean` | `Boolean` | |
| `null` | `null` | |
| `Promise<T>` | `com.facebook.react.bridge.Promise` | Async result |
| `Object` (plain) | `ReadableMap` | Key-value map |
| `Array<T>` | `ReadableArray` | |
| `(args) => void` | `Callback` | Event/callback |

## Step 2: Configure Codegen

In `package.json`, register the spec location:

```json
{
  "name": "your-app",
  "codegenConfig": {
    "name": "AppSpecs",
    "type": "modules",
    "jsSrcsDir": "src/specs",
    "android": {
      "javaPackageName": "com.yourapp"
    }
  }
}
```

Run Codegen:

```bash
cd android && ./gradlew generateCodegenArtifactsFromSchema
```

This generates (inside `android/app/build/generated/source/codegen/`):

```
NativeDeviceInfoModuleSpec.java  ŌåÉ abstract class your Kotlin class extends
NativeDeviceInfoModuleSpecJSI.cpp ŌåÉ C++ JSI binding (do not edit)
```

## Step 3: Kotlin Implementation

Create `android/app/src/main/java/com/yourapp/NativeDeviceInfoModule.kt`:

```kotlin
package com.yourapp

import android.os.BatteryManager
import android.os.Build
import android.content.Context
import android.net.ConnectivityManager
import android.net.NetworkCapabilities
import com.facebook.react.bridge.Promise
import com.facebook.react.bridge.ReactApplicationContext
import com.yourapp.NativeDeviceInfoModuleSpec  // generated by Codegen

class NativeDeviceInfoModule(
    reactContext: ReactApplicationContext
) : NativeDeviceInfoModuleSpec(reactContext) {

    // The name must match the spec registration key
    override fun getName(): String = "NativeDeviceInfoModule"

    // Synchronous ŌĆö returns directly to JS via JSI
    override fun getDeviceModel(): String {
        return "${Build.MANUFACTURER} ${Build.MODEL}"
    }

    override fun getAndroidVersion(): Double {
        return Build.VERSION.SDK_INT.toDouble()
    }

    // Async ŌĆö resolve/reject the Promise
    override fun getBatteryLevel(promise: Promise) {
        try {
            val bm = reactApplicationContext
                .getSystemService(Context.BATTERY_SERVICE) as BatteryManager
            val level = bm.getIntProperty(BatteryManager.BATTERY_PROPERTY_CAPACITY)
            promise.resolve(level.toDouble())
        } catch (e: Exception) {
            promise.reject("BATTERY_ERROR", e.message, e)
        }
    }

    override fun getNetworkType(promise: Promise) {
        try {
            val cm = reactApplicationContext
                .getSystemService(Context.CONNECTIVITY_SERVICE) as ConnectivityManager
            val network = cm.activeNetwork
            val caps = cm.getNetworkCapabilities(network)

            val type = when {
                caps == null -> "none"
                caps.hasTransport(NetworkCapabilities.TRANSPORT_WIFI) -> "wifi"
                caps.hasTransport(NetworkCapabilities.TRANSPORT_CELLULAR) -> "cellular"
                else -> "none"
            }
            promise.resolve(type)
        } catch (e: Exception) {
            promise.reject("NETWORK_ERROR", e.message, e)
        }
    }

    override fun readFile(path: String, promise: Promise) {
        try {
            val content = java.io.File(path).readText()
            promise.resolve(content)
        } catch (e: Exception) {
            promise.reject("FILE_READ_ERROR", "Cannot read $path: ${e.message}", e)
        }
    }

    override fun writeFile(path: String, content: String, promise: Promise) {
        try {
            java.io.File(path).writeText(content)
            promise.resolve(true)
        } catch (e: Exception) {
            promise.reject("FILE_WRITE_ERROR", "Cannot write $path: ${e.message}", e)
        }
    }

    override fun subscribeToOrientationChanges(callback: com.facebook.react.bridge.Callback) {
        // Store callback, call it when orientation changes
        // (omitted for brevity ŌĆö use ReactContext event emitter in practice)
    }
}
```

## Step 4: Register the Module

Create `android/app/src/main/java/com/yourapp/DeviceInfoPackage.kt`:

```kotlin
package com.yourapp

import com.facebook.react.TurboReactPackage
import com.facebook.react.bridge.ReactApplicationContext
import com.facebook.react.module.model.ReactModuleInfo
import com.facebook.react.module.model.ReactModuleInfoProvider

class DeviceInfoPackage : TurboReactPackage() {

    override fun getModule(name: String, context: ReactApplicationContext) =
        if (name == NativeDeviceInfoModule.NAME) NativeDeviceInfoModule(context)
        else null

    override fun getReactModuleInfoProvider() = ReactModuleInfoProvider {
        mapOf(
            NativeDeviceInfoModule.NAME to ReactModuleInfo(
                NativeDeviceInfoModule.NAME,
                NativeDeviceInfoModule::class.java.name,
                false,  // canOverrideExistingModule
                false,  // needsEagerInit
                true,   // isCxxModule
                true    // isTurboModule ŌĆö THIS IS THE KEY FLAG
            )
        )
    }

    companion object {
        const val NAME = "NativeDeviceInfoModule"
    }
}
```

Register in `MainApplication.kt`:

```kotlin
override fun getPackages(): List<ReactPackage> =
    PackageList(this).packages.apply {
        add(DeviceInfoPackage())
    }
```

## Step 5: Use from JavaScript/TypeScript

```tsx
// hooks/useDeviceInfo.ts
import NativeDeviceInfo from '../specs/NativeDeviceInfoModule';
import { useState, useEffect } from 'react';

export function useDeviceInfo() {
  const [batteryLevel, setBatteryLevel] = useState<number | null>(null);
  const [networkType, setNetworkType] = useState<string | null>(null);

  // Synchronous calls ŌĆö no await needed
  const model = NativeDeviceInfo.getDeviceModel();
  const androidVersion = NativeDeviceInfo.getAndroidVersion();

  useEffect(() => {
    // Async calls
    NativeDeviceInfo.getBatteryLevel().then(setBatteryLevel);
    NativeDeviceInfo.getNetworkType().then(setNetworkType);
  }, []);

  return { model, androidVersion, batteryLevel, networkType };
}

// In a component
function DeviceInfoScreen() {
  const info = useDeviceInfo();

  return (
    <View>
      <Text>Model: {info.model}</Text>
      <Text>Android: {info.androidVersion}</Text>
      <Text>Battery: {info.batteryLevel}%</Text>
      <Text>Network: {info.networkType}</Text>
    </View>
  );
}
```

## TurboModule with Coroutines

The Promise-based API works but Kotlin coroutines are more idiomatic. Use `kotlinx-coroutines-android` to bridge them:

```kotlin
// build.gradle (app)
implementation("org.jetbrains.kotlinx:kotlinx-coroutines-android:1.7.3")
```

```kotlin
import kotlinx.coroutines.*

override fun getBatteryLevel(promise: Promise) {
    CoroutineScope(Dispatchers.IO).launch {
        try {
            val level = fetchBatteryLevelSuspend()  // your suspend fun
            promise.resolve(level)
        } catch (e: Exception) {
            promise.reject("BATTERY_ERROR", e)
        }
    }
}

private suspend fun fetchBatteryLevelSuspend(): Double = withContext(Dispatchers.Main) {
    val bm = reactApplicationContext
        .getSystemService(Context.BATTERY_SERVICE) as BatteryManager
    bm.getIntProperty(BatteryManager.BATTERY_PROPERTY_CAPACITY).toDouble()
}
```

## Defensive Loading: getEnforcing vs get

```typescript
// getEnforcing ŌĆö throws if the module is not available
// Use when the module is required for the app to function
export default TurboModuleRegistry.getEnforcing<Spec>('NativeDeviceInfoModule');

// get ŌĆö returns null if the module is not available
// Use for optional modules or when supporting both architectures
const NativeDeviceInfo = TurboModuleRegistry.get<Spec>('NativeDeviceInfoModule');

// Guard in your hook
function useDeviceInfo() {
  if (!NativeDeviceInfo) {
    return { model: 'unknown', androidVersion: 0 };
  }
  return {
    model: NativeDeviceInfo.getDeviceModel(),
    androidVersion: NativeDeviceInfo.getAndroidVersion(),
  };
}
```
